🐳 4种部署方式
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. 📦 脚本部署 (推荐)
   bash scripts/install.sh && bash scripts/start.sh

2. 🐋 Docker 部署
   docker-compose up -d

3. ☸️  Kubernetes 部署
   kubectl apply -f k8s/

4. 🔧 手动部署
   python3 -m venv venv && source venv/bin/activate
   pip install -r requirements.txt && python src/main.py
